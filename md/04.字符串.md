# 字符串

```swift
import Foundation

/**
 字符串
 */

// 1. 字面量创建
var str = "123"
print("\(str)")

// 2. String 装饰器

var str1 = String("123")

print("\(str1)")

// 3. 判断是否为空串

var str2 = ""

print("\(str2.isEmpty)")

// 4. 多行字面量
// 其中行尾加 '\', 表示不换行
var str3 = """
这是第1行   \
这是第2行
    这是第3行
"""

print("\(str3)")
/**
 常见特殊字符
 \0（空字符）、\\（反斜杠）、\t（制表符）、\n（换行符）、\r（回车符）、\"（双引号）、\'（单引号）
 */

```
