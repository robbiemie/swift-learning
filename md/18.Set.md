## Set

```swift
/**
 Set
 */
// 初始化 Set
var set: Set = ["aaa", "bbb", "ccc"]

print(set) // 输出结果是无序

// 插入一个值
set.insert("ddd")

print(set)

// 插入一个类

struct Person {
    var name: String
    var age: Int
}

extension Person: Hashable {
    func hash(into hasher: inout Hasher) {
        hasher.combine(age)
        hasher.combine(name)
    }
}

let personSet:Set<Person> = [Person(name: "aaa", age: 19)]
print(personSet)

```
